class DataManager:
    """
    This class manages and preprocesses a simple text dataset for a sentence classification task.

    Attributes:
        verbose (bool): Controls verbosity for printing information during data processing.
        max_sentence_len (int): The maximum length of a sentence in the dataset.
        str_questions (list): A list to store the string representations of the questions in the dataset.
        str_labels (list): A list to store the string representations of the labels in the dataset.
        numeral_labels (list): A list to store the numerical representations of the labels in the dataset.
        numeral_data (list): A list to store the numerical representations of the questions in the dataset.
        random_state (int): Seed value for random number generation to ensure reproducibility.
            Set this value to a specific integer to reproduce the same random sequence every time. Defaults to 6789.
        random (np.random.RandomState): Random number generator object initialized with the given random_state.
            It is used for various random operations in the class.

    Methods:
        maybe_download(dir_name, file_name, url, verbose=True):
            Downloads a file from a given URL if it does not exist in the specified directory.
            The directory and file are created if they do not exist.

        read_data(dir_name, file_names):
            Reads data from files in a directory, preprocesses it, and computes the maximum sentence length.
            Each file is expected to contain rows in the format "<label>:<question>".
            The labels and questions are stored as string representations.

        manipulate_data():
            Performs data manipulation by tokenizing, numericalizing, and padding the text data.
            The questions are tokenized and converted into numerical sequences using a tokenizer.
            The sequences are padded or truncated to the maximum sequence length.

        train_valid_test_split(train_ratio=0.9):
            Splits the data into training, validation, and test sets based on a given ratio.
            The data is randomly shuffled, and the specified ratio is used to determine the size of the training set.
            The string questions, numerical data, and numerical labels are split accordingly.
            TensorFlow `Dataset` objects are created for the training and validation sets.


    """

        if not os.path.exists(dir_name):

        if not os.path.exists(os.path.join(dir_name, file_name)):
            urlretrieve(url + file_name, os.path.join(dir_name, file_name))

            print("Downloaded successfully {}".format(file_name))

    def read_data(self, dir_name, file_names):
        self.str_questions = list()
        self.str_labels = list()
        for file_name in file_names:
            file_path= os.path.join(dir_name, file_name)
            with open(file_path, "r", encoding="latin-1") as f:
                for row in f:
                    row_str = row.split(":")
                    label, question = row_str[0], row_str[1]
                    question = question.lower()
                    self.str_labels.append(label)
                    self.str_questions.append(question[0:-1])
                    if self.max_sentence_len < len(self.str_questions[-1]):
                        self.max_sentence_len = len(self.str_questions[-1])

        # turns labels into numbers
        le = preprocessing.LabelEncoder()
        le.fit(self.str_labels)

        self.str_classes = le.classes_
        self.num_classes = len(self.str_classes)

class BaseTrainer:
    def __init__(self, model, criterion, optimizer, train_loader, val_loader):
        self.model = model
        self.criterion = criterion  #the loss function
        self.optimizer = optimizer  #the optimizer
        self.train_loader = train_loader  #the train loader
        self.val_loader = val_loader  #the valid loader

    #the function to train the model in many epochs
    def fit(self, num_epochs):
        self.num_batches = len(self.train_loader)

        for epoch in range(num_epochs):
            print(f'Epoch {epoch + 1}/{num_epochs}')
            train_loss, train_accuracy = self.train_one_epoch()
            val_loss, val_accuracy = self.validate_one_epoch()
            print(
                f'{self.num_batches}/{self.num_batches} - train_loss: {train_loss:.4f} - train_accuracy: {train_accuracy*100:.4f}% \
                - val_loss: {val_loss:.4f} - val_accuracy: {val_accuracy*100:.4f}%')

    #train in one epoch, return the train_acc, train_loss
    def train_one_epoch(self):
        self.model.train()
        running_loss, correct, total = 0.0, 0, 0
        for i, data in enumerate(self.train_loader):
            inputs, labels = data
            inputs, labels = inputs.to(device), labels.to(device)
            self.optimizer.zero_grad()
            outputs = self.model(inputs)
            loss = self.criterion(outputs, labels)
            loss.backward()
            self.optimizer.step()

            running_loss += loss.item()
            _, predicted = torch.max(outputs.data, 1)
            total += labels.size(0)
            correct += (predicted == labels).sum().item()
        train_accuracy = correct / total
        train_loss = running_loss / self.num_batches
        return train_loss, train_accuracy

    #evaluate on a loader and return the loss and accuracy
    def evaluate(self, loader):
        self.model.eval()
        loss, correct, total = 0.0, 0, 0
        with torch.no_grad():
            for data in loader:
                inputs, labels = data
                inputs, labels = inputs.to(device), labels.to(device)
                outputs = self.model(inputs)
                loss = self.criterion(outputs, labels)
                loss += loss.item()
                _, predicted = torch.max(outputs.data, 1)
                total += labels.size(0)
                correct += (predicted == labels).sum().item()

        accuracy = correct / total
        loss = loss / len(self.val_loader)
        return loss, accuracy

    #return the val_acc, val_loss, be called at the end of each epoch
    def validate_one_epoch(self):
      val_loss, val_accuracy = self.evaluate(self.val_loader)
      return val_loss, val_accuracy

class BaseRNN(nn.Module):
    def __init__(self, cell_type='gru', embed_size=128, state_sizes=[128, 128],
                 output_type="mean", data_manager=None,run_mode='scratch', pretrained_weight=None):
        super().__init__()
        assert data_manager is not None, "Provide data_manager=dm"
        assert output_type in {"last_state", "mean", "max"}
        assert run_mode in {"scratch", "init-only", "init+fine-tune"} 


        # required attributes
        self.data_manager = data_manager
        self.cell_type    = cell_type
        self.state_sizes  = list(state_sizes)
        self.output_type  = output_type
        self.embed_size   = embed_size

        # derived
        self.vocab_size   = data_manager.vocab_size
        self.num_classes  = data_manager.num_classes
        self.pad_idx      = getattr(getattr(data_manager, "tokenizer", None), "pad_token_id", 0)

        # build the modules
        self.build()


    # return the corresponding memory cell
    @staticmethod
    def get_layer(cell_type='gru', input_size=128, state_size=128):
        if cell_type == 'gru':
            return nn.GRU(input_size, state_size, batch_first=True)
        elif cell_type == 'lstm':
            return nn.LSTM(input_size, state_size, batch_first=True)
        elif cell_type == 'simple_rnn':
            return nn.RNN(input_size, state_size, nonlinearity='tanh', batch_first=True)
        else:
            raise ValueError(f"Unknown cell_type: {cell_type}")

    def get_cell(self, *args, **kwargs):
        return BaseRNN.get_layer(*args, **kwargs)
    
    def build(self):
        # embedding layer (padding aware)
        self.embedding = nn.Embedding(self.vocab_size, self.embed_size, padding_idx=self.pad_idx)

        # recurrent stack
        self.rnns = nn.ModuleList()
        in_size = self.embed_size
        for hs in self.state_sizes:
            self.rnns.append(self.get_cell(self.cell_type, input_size=in_size, state_size=hs))

            in_size = hs  # output size of this layer is input to next

        # classification head, last hidden size == state_sizes[-1]
        self.fc = nn.Linear(in_size, self.num_classes)


    def forward(self, x):
        out = self.embedding(x)              # [B,T,E]
        for rnn in self.rnns:
            out, _ = rnn(out)                # [B,T,H]

        # pooling according to output_type
        mask = (x != self.pad_idx).unsqueeze(-1)       # [B,T,1]
        lengths = mask.sum(dim=1).squeeze(-1)          # [B]

        if self.output_type == "mean":
            pooled = (out * mask.float()).sum(dim=1) / \
                     lengths.clamp(min=1).unsqueeze(1).float()
        elif self.output_type == "max":
            pooled = out.masked_fill(~mask, float('-inf')).max(dim=1).values
            pooled[~torch.isfinite(pooled)] = 0.0
        else:  # last_state
            B, T, H = out.size()
            idx = (lengths - 1).clamp(min=0).view(B,1,1).expand(B,1,H)
            pooled = out.gather(dim=1, index=idx).squeeze(1)

        return self.fc(pooled)

    def build_embedding_matrix(self):
        # Insert your code here. Your code should allow for saving the embedding matrix in ``self.embed_path'' (as numpy array) for future retrieval.
        # Use cache if present and the shape matches
        if os.path.exists(self.embed_path):
            mat = np.load(self.embed_path)
            if mat.shape == (self.vocab_size, self.embed_size):
                self.embed_matrix = mat.astype(np.float32)
                return self.embed_matrix

        kv = api.load(self.embed_model)  # gensim KeyedVectors

        if 0 <= self.pad_idx < self.vocab_size:

        key_to_index = getattr(kv, "key_to_index", {})
        for token, idx in self.word2idx.items():
            if token in key_to_index:
                mat[idx] = kv[token]

        np.save(self.embed_path, mat)  # cache for later runs
        self.embed_matrix = mat
        return self.embed_matrix

3.3. RNNs with Attention for Text and Sequence Classification

In what follows, you are required to implement a RNN with the attention mechanism for text and sequence classification. This attention mechanism is applied at the last hidden layer of our RNN. Specifically, let $\textbf{h}_1^L, \textbf{h}_2^L,...,\textbf{h}_{T-1}^L, \textbf{h}_T^L$ be the hidden states at the last hidden layer $L$ where $T$ is the sequence length. We compute the context vector $\textbf{c}$ as $\textbf{c}=\sum_{i=1}^{T}\textbf{a}_{i}\textbf{h}_{i}^L$ where $\textbf{a}_1,...,\textbf{a}_T$ are the alignment weights (i.e., $\textbf{a}_i\geq 0$ and $\sum_{i=1}^{T}\textbf{a}_{i}=1$).

The alignment weights are computed as follows:

  • $\textbf{a}=[\textbf{a}_{i}]_{i=1}^{T}=\text{softmax}(\textbf{s})$ where $\textbf{s}= [\textbf{s}_{i}]_{i=1}^{T}$ consists of the alignment scores.
  • The alignment scores are computed as $\textbf{s}=\text{tanh}(\textbf{h}^LU)V$ where $\textbf{h}^L=\left[\begin{array}{c} \textbf{h}_{1}^L\\ \textbf{h}_{2}^L\\ ...\\ \textbf{h}_{T-1}^L\\ \textbf{h}_{T}^L \end{array}\right]\in\mathbb{R}^{T\times state\_size_{L}}$, $U\in\mathbb{R}^{state\_size_{L}\times output\_length}$, $V\in\mathbb{R}^{output\_length\times1}$, and $output\_length$ is a hyperparameter. Note that if we consider a mini-batch, the shape of $\textbf{h}^L$ is $(batch\_size, T, state\_size_L)$ where $state\_size_L$ is the hidden size of the last hidden layer. The figure on the right below illustrates the process of calculating a score $\textbf{s}_i$ for an individual hidden state $\textbf{h}_i^L$. Weight matrices $U$ and $V$ are shared across the hidden states $\textbf{h}_1^L,\textbf{h}_2^L,\dots,\textbf{h}_T^L$.

After having the context vector $\textbf{c}$, we concatenate with the output hidden state which is computed by aggregating $\textbf{h}^{1:T}_L$ following $3$ different output_type as above. On top of this concatenation, we conduct the output layer with the softmax activation.

class MyAttention(nn.Module):
    def __init__(self, output_length=50):
        super().__init__()
        # Insert your code here

        self.U = nn.Parameter(torch.empty(state_size, output_length))
        self.V = nn.Parameter(torch.empty(output_length, 1))
        nn.init.xavier_uniform_(self.U)
        nn.init.xavier_uniform_(self.V)

    def forward(self, all_states, last_state):
        B, T, H = all_states.shape

        # Compute scores
        proj = torch.tanh(torch.matmul(all_states, self.U))     # (B, T, output_length)
        scores = torch.matmul(proj, self.V).squeeze(-1)          # (B, T)

        # Attention weights
        attn = F.softmax(scores, dim=1).unsqueeze(-1)            # (B, T, 1)

        # Context vector
        context_vector = torch.sum(attn * all_states, dim=1)            # (B, H)
        return context_vector

class MyAttention(nn.Module):
    def __init__(self, output_length: int = 50):
        super().__init__()
        # Projects last hidden dim H  M lazily on first forward (no shape mismatch)
        self.U = nn.LazyLinear(output_length, bias=False)
        self.V = nn.Linear(output_length, 1, bias=False)

    def forward(self, all_states: torch.Tensor, last_state: torch.Tensor):
        # all_states: (B, T, H)
        proj   = torch.tanh(self.U(all_states))      # (B, T, M)
        scores = self.V(proj).squeeze(-1)            # (B, T)
        attn   = F.softmax(scores, dim=1).unsqueeze(-1)
        context = torch.sum(attn * all_states, dim=1)  # (B, H)
        return context

class AttentionRNN(BaseRNN):
    def __init__(self,
                 cell_type: str = "gru",
                 embed_size: int = 128,
                 state_sizes: list[int] | None = None,
                 output_type: str = "mean",
                 data_manager=None,
                 attn_output_length: int = 50):
        self.attn_output_length = attn_output_length   # set BEFORE super()
        if state_sizes is None:
            state_sizes = [128, 128]
        super().__init__(cell_type, embed_size, state_sizes, output_type, data_manager)

    def build(self):
        # ---- vocab & pad setup ----
        tok = getattr(self.data_manager, "tokenizer", None)
        vocab_size = getattr(self.data_manager, "vocab_size", None)
        if vocab_size is None:
            if tok is None:
                raise ValueError(
                    "DataManager must provide either vocab_size or tokenizer before building the model."
                )
            vocab_size = len(tok.get_vocab())

        pad_id = getattr(self.data_manager, "pad_id", getattr(tok, "pad_token_id", 0))
        self.pad_id = pad_id  # store for masking later

        # ---- embedding ----
        self.embedding = nn.Embedding(
            num_embeddings=vocab_size,
            embedding_dim=self.embed_size,   # use embed_size (not embed_dim)
            padding_idx=self.pad_id,
        )

        # ---- recurrent core (last element is hidden size; num layers = len(state_sizes)) ----
        hidden_size = self.state_sizes[-1]
        num_layers = len(self.state_sizes)

        rnn_map = {"gru": nn.GRU, "lstm": nn.LSTM, "rnn": nn.RNN}
        RNN = rnn_map[self.cell_type.lower()]
        self.rnn = RNN(
            input_size=self.embed_size,
            hidden_size=hidden_size,
            num_layers=num_layers,
            batch_first=True,
            bidirectional=False,
        )

        # ---- attention & classifier ----
        self.attention = MyAttention(output_length=self.attn_output_length)
        self.dropout = nn.Dropout(p=0.3)
        self.classifier = nn.Linear(2 * hidden_size, self.data_manager.num_classes)

    def _get_mask_and_lengths(self, x):
        pad_id = getattr(self.data_manager, "pad_id", 0)
        mask = (x != pad_id)              # (B, T)
        lengths = mask.sum(dim=1)         # (B,)
        return mask, lengths

    def forward(self, x):
        # x: (B, T) token ids
        mask, lengths = self._get_mask_and_lengths(x)

        emb = self.embedding(x)                           # (B, T, E)
        outputs, _ = self.rnn(emb)                        # (B, T, H*)  <-- LAST LAYER sequence

        # last valid state from LAST layer sequence
        idx = (lengths - 1).clamp(min=0)
        last_state = outputs[torch.arange(outputs.size(0), device=outputs.device), idx, :]  # (B, H*)

        # attention over LAST layer sequence
        context = self.attention(outputs, last_state)     # (B, H*)

        # aggregate LAST layer sequence
        if self.output_type == "last":
            pooled = last_state
        elif self.output_type == "max":
            neg_inf = torch.finfo(outputs.dtype).min
            pooled, _ = outputs.masked_fill(~mask.unsqueeze(-1), neg_inf).max(dim=1)
        else:  # "mean"
            summed = (outputs * mask.unsqueeze(-1)).sum(dim=1)
            pooled = summed / lengths.unsqueeze(1).clamp(min=1)

        logits = self.classifier(self.dropout(torch.cat([context, pooled], dim=1)))  # (B, C)
        return logits
